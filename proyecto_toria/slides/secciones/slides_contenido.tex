\section{Introducción}
\begin{frame}{El Desafío de la Logística Hospitalaria}
    \begin{itemize}
        \item Creciente demanda de servicios asistenciales.
        \item Necesidad de liberar al personal de tareas repetitivas.
        \item Solución: Cobots móviles para transporte de medicinas y comida.
    \end{itemize}
\end{frame}

\section{Simulación y ROS 2}
\begin{frame}{Por qué ROS 2 y Simulación}
    \begin{itemize}
        \item \textbf{ROS 2:} Estándar industrial, modularidad y seguridad.
        \item \textbf{Entornos:} AWS RoboMaker Hospital, Open-RMF.
        \item \textbf{Validación:} Pruebas de navegación segura sin riesgos físicos.
    \end{itemize}
\end{frame}

\section{Diseño del Robot}
\begin{frame}{Características Técnicas y Materiales}
    \begin{block}{Materiales}
        Antimicrobianos, resistentes a desinfección química (ISO 13482).
    \end{block}
    \begin{block}{Sensores Clave}
        \begin{itemize}
            \item LiDAR 3D (Mapeo y SLAM).
            \item Cámaras RGB-D (Detección de obstáculos).
            \item Ultrasonidos (Seguridad redundante).
        \end{itemize}
    \end{block}
\end{frame}

\section{Infraestructura}
\begin{frame}{Dependencias de Infraestructura}
    \begin{itemize}
        \item \textbf{Carga Automática:} Salas de docking estaciones.
        \item \textbf{Desinfección:}
            \begin{itemize}
                \item Cámaras de luz UV-C dedicadas.
                \item Integración con Open-RMF para gestión de puertas/ascensores.
            \end{itemize}
    \end{itemize}
\end{frame}

\section{Conclusiones}
\begin{frame}{Futuro de la Robótica de Servicio}
    \begin{itemize}
        \item Los robots no sustituyen, sino que \textbf{asisten}.
        \item La estandarización (ISO 13482) es crucial.
        \item El éxito depende de la integración con el entorno hospitalario.
    \end{itemize}
\end{frame}
